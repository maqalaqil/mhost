# mhost API Layer Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a feature-flagged public API layer to mhost with REST endpoints, WebSocket streaming, token auth, and outbound webhooks — all running inside the daemon process.

**Architecture:** New `mhost-api` crate with axum HTTP/WS server that shares the daemon's `Arc<Supervisor>` and `Arc<Mutex<StateStore>>` directly. Gated behind `--features api` Cargo feature flag so the default binary stays lean. Event bus via `tokio::sync::broadcast` connects supervisor events to WebSocket clients and webhook dispatch.

**Tech Stack:** axum 0.7, tokio-tungstenite (via axum), argon2 (token hashing), tower-http (CORS), hmac+sha2 (webhook signing, reuse existing pattern from mhost-notify)

**Spec:** `docs/superpowers/specs/2026-04-04-api-layer-design.md`

```rust
use axum::http::StatusCode;
use axum::response::{IntoResponse, Response};
use serde::Serialize;

#[derive(Debug, Serialize)]
pub struct ApiResponse<T: Serialize> {
    pub ok: bool,
    #[serde(skip_serializing_if = "Option::is_none")]
    pub data: Option<T>,
    #[serde(skip_serializing_if = "Option::is_none")]
    pub error: Option<String>,
}

impl<T: Serialize> ApiResponse<T> {
    pub fn success(data: T) -> Self {
        Self {
            ok: true,
            data: Some(data),
            error: None,
        }
    }
}

impl ApiResponse<()> {
    pub fn error(message: impl Into<String>) -> Self {
        Self {
            ok: false,
            data: None,
            error: Some(message.into()),
        }
    }

    pub fn ok() -> Self {
        Self {
            ok: true,
            data: None,
            error: None,
        }
    }
}

impl<T: Serialize> IntoResponse for ApiResponse<T> {
    fn into_response(self) -> Response {
        let status = if self.ok {
            StatusCode::OK
        } else {
            StatusCode::BAD_REQUEST
        };
        let body = serde_json::to_string(&self).unwrap_or_else(|_| {
            r#"{"ok":false,"error":"Serialization error"}"#.to_string()
        });
        (status, [("content-type", "application/json")], body).into_response()
    }
}

pub struct ApiError {
    pub status: StatusCode,
    pub message: String,
}

impl ApiError {
    pub fn not_found(msg: impl Into<String>) -> Self {
        Self { status: StatusCode::NOT_FOUND, message: msg.into() }
    }
    pub fn unauthorized(msg: impl Into<String>) -> Self {
        Self { status: StatusCode::UNAUTHORIZED, message: msg.into() }
    }
    pub fn forbidden(msg: impl Into<String>) -> Self {
        Self { status: StatusCode::FORBIDDEN, message: msg.into() }
    }
    pub fn internal(msg: impl Into<String>) -> Self {
        Self { status: StatusCode::INTERNAL_SERVER_ERROR, message: msg.into() }
    }
}

impl IntoResponse for ApiError {
    fn into_response(self) -> Response {
        let body = serde_json::json!({
            "ok": false,
            "error": self.message,
        });
        (self.status, [("content-type", "application/json")], body.to_string()).into_response()
    }
}
```

```rust
use chrono::{DateTime, Utc};
use serde::Serialize;
use tokio::sync::broadcast;

#[derive(Debug, Clone, Serialize)]
pub struct ProcessEvent {
    pub event: String,
    pub process: String,
    pub timestamp: DateTime<Utc>,
    pub data: serde_json::Value,
}

impl ProcessEvent {
    pub fn new(event: impl Into<String>, process: impl Into<String>, data: serde_json::Value) -> Self {
        Self {
            event: event.into(),
            process: process.into(),
            timestamp: Utc::now(),
            data,
        }
    }
}

#[derive(Clone)]
pub struct EventBus {
    sender: broadcast::Sender<ProcessEvent>,
}

impl EventBus {
    pub fn new(capacity: usize) -> Self {
        let (sender, _) = broadcast::channel(capacity);
        Self { sender }
    }

    pub fn publish(&self, event: ProcessEvent) {
        // Ignore error (no subscribers is fine)
        let _ = self.sender.send(event);
    }

    pub fn subscribe(&self) -> broadcast::Receiver<ProcessEvent> {
        self.sender.subscribe()
    }
}

impl Default for EventBus {
    fn default() -> Self {
        Self::new(1024)
    }
}
```

```rust
use argon2::{Argon2, PasswordHasher, PasswordVerifier};
use argon2::password_hash::{rand_core::OsRng, SaltString, PasswordHash};
use chrono::{DateTime, Utc};
use serde::{Deserialize, Serialize};
use std::path::Path;
use uuid::Uuid;

use crate::roles::Role;

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct ApiToken {
    pub id: String,
    pub name: String,
    pub secret_hash: String,
    pub role: Role,
    pub created_at: DateTime<Utc>,
    pub last_used: Option<DateTime<Utc>>,
    pub expires_at: Option<DateTime<Utc>>,
}

impl ApiToken {
    pub fn is_expired(&self) -> bool {
        self.expires_at
            .map(|exp| Utc::now() > exp)
            .unwrap_or(false)
    }
}

pub struct TokenStore {
    path: std::path::PathBuf,
    tokens: Vec<ApiToken>,
}

pub struct CreatedToken {
    pub token: ApiToken,
    pub raw_secret: String,
}

impl TokenStore {
    pub fn load(path: &Path) -> Result<Self, String> {
        let tokens = if path.exists() {
            let data = std::fs::read_to_string(path)
                .map_err(|e| format!("Failed to read token store: {e}"))?;
            serde_json::from_str(&data)
                .map_err(|e| format!("Failed to parse token store: {e}"))?
        } else {
            Vec::new()
        };
        Ok(Self { path: path.to_path_buf(), tokens })
    }

    pub fn save(&self) -> Result<(), String> {
        let data = serde_json::to_string_pretty(&self.tokens)
            .map_err(|e| format!("Failed to serialize tokens: {e}"))?;
        std::fs::write(&self.path, data)
            .map_err(|e| format!("Failed to write token store: {e}"))?;
        Ok(())
    }

    pub fn create(&mut self, name: &str, role: Role, expires_at: Option<DateTime<Utc>>) -> Result<CreatedToken, String> {
        if self.tokens.iter().any(|t| t.name == name) {
            return Err(format!("Token with name '{name}' already exists"));
        }

        let id = format!("tok_{}", &Uuid::new_v4().to_string()[..8]);
        let raw_secret = format!("mhost_{id}_{}", Uuid::new_v4().to_string().replace('-', ""));
        let secret_hash = hash_secret(&raw_secret)?;

        let token = ApiToken {
            id,
            name: name.to_string(),
            secret_hash,
            role,
            created_at: Utc::now(),
            last_used: None,
            expires_at,
        };

        self.tokens.push(token.clone());
        self.save()?;

        Ok(CreatedToken { token, raw_secret })
    }

    pub fn verify(&self, raw_secret: &str) -> Option<&ApiToken> {
        self.tokens.iter().find(|t| {
            if t.is_expired() {
                return false;
            }
            verify_secret(raw_secret, &t.secret_hash)
        })
    }

    pub fn update_last_used(&mut self, id: &str) {
        if let Some(token) = self.tokens.iter_mut().find(|t| t.id == id) {
            token.last_used = Some(Utc::now());
            let _ = self.save();
        }
    }

    pub fn list(&self) -> &[ApiToken] {
        &self.tokens
    }

    pub fn revoke(&mut self, id: &str) -> Result<(), String> {
        let len_before = self.tokens.len();
        self.tokens.retain(|t| t.id != id);
        if self.tokens.len() == len_before {
            return Err(format!("Token '{id}' not found"));
        }
        self.save()
    }
}

fn hash_secret(secret: &str) -> Result<String, String> {
    let salt = SaltString::generate(&mut OsRng);
    let argon2 = Argon2::default();
    let hash = argon2
        .hash_password(secret.as_bytes(), &salt)
        .map_err(|e| format!("Hash error: {e}"))?;
    Ok(hash.to_string())
}

fn verify_secret(secret: &str, hash: &str) -> bool {
    let parsed = match PasswordHash::new(hash) {
        Ok(h) => h,
        Err(_) => return false,
    };
    Argon2::default().verify_password(secret.as_bytes(), &parsed).is_ok()
}
```

```rust
use axum::extract::Request;
use axum::middleware::Next;
use axum::response::Response;
use std::sync::Arc;
use tokio::sync::Mutex;

use crate::envelope::ApiError;
use crate::rate_limit::RateLimiter;
use crate::roles::Role;
use crate::tokens::TokenStore;

#[derive(Clone)]
pub struct AuthState {
    pub token_store: Arc<Mutex<TokenStore>>,
    pub rate_limiter: Arc<Mutex<RateLimiter>>,
}

#[derive(Clone)]
pub struct AuthenticatedUser {
    pub token_id: String,
    pub name: String,
    pub role: Role,
}

pub async fn auth_middleware(
    axum::extract::State(state): axum::extract::State<AuthState>,
    mut req: Request,
    next: Next,
) -> Result<Response, ApiError> {
    let raw_token = extract_bearer(&req)
        .ok_or_else(|| ApiError::unauthorized("Missing or invalid Authorization header. Use: Bearer <token>"))?;

    let mut store = state.token_store.lock().await;
    let token = store.verify(raw_token)
        .ok_or_else(|| ApiError::unauthorized("Invalid or expired API token"))?;

    let user = AuthenticatedUser {
        token_id: token.id.clone(),
        name: token.name.clone(),
        role: token.role,
    };

    // Rate limit check
    {
        let mut limiter = state.rate_limiter.lock().await;
        if !limiter.check(&user.token_id) {
            return Err(ApiError {
                status: axum::http::StatusCode::TOO_MANY_REQUESTS,
                message: "Rate limit exceeded".to_string(),
            });
        }
    }

    store.update_last_used(&user.token_id);
    drop(store);

    req.extensions_mut().insert(user);
    Ok(next.run(req).await)
```

```rust
use std::collections::HashMap;
use std::time::{Duration, Instant};

pub struct RateLimiter {
    max_per_window: u32,
    window: Duration,
    buckets: HashMap<String, Vec<Instant>>,
}

impl RateLimiter {
    pub fn new(max_per_window: u32, window: Duration) -> Self {
        Self {
            max_per_window,
            window,
            buckets: HashMap::new(),
        }
    }

    pub fn check(&mut self, key: &str) -> bool {
        let now = Instant::now();
        let cutoff = now - self.window;

        let bucket = self.buckets.entry(key.to_string()).or_default();
        bucket.retain(|t| *t > cutoff);

        if bucket.len() as u32 >= self.max_per_window {
            return false;
        }

        bucket.push(now);
        true
    }
}
```

```rust
pub async fn start_api_server(
    bind: SocketAddr,
    paths: MhostPaths,
    supervisor: Arc<dyn SupervisorApi + Send + Sync>,
    event_bus: EventBus,
) -> Result<(), String> {
    let token_store = TokenStore::load(&paths.api_tokens())
        .map_err(|e| format!("Failed to load token store: {e}"))?;

    let rate_limiter = RateLimiter::new(100, std::time::Duration::from_secs(60));

    let auth_state = AuthState {
        token_store: Arc::new(Mutex::new(token_store)),
        rate_limiter: Arc::new(Mutex::new(rate_limiter)),
    };

    let webhook_dispatcher = Arc::new(
        WebhookDispatcher::load(&paths.webhooks_config(), &paths.webhook_failures())
            .unwrap_or_default()
    );

    let state = AppState {
        auth: auth_state.clone(),
        event_bus: event_bus.clone(),
        paths,
        webhook_dispatcher: Arc::clone(&webhook_dispatcher),
        supervisor,
    };

    // Start webhook event forwarder
    let wd = Arc::clone(&webhook_dispatcher);
    let mut event_rx = event_bus.subscribe();
    tokio::spawn(async move {
        while let Ok(event) = event_rx.recv().await {
            wd.dispatch(&event).await;
        }
    });

    let cors = CorsLayer::new()
        .allow_origin(Any)
        .allow_methods(Any)
        .allow_headers(Any);

    let public_routes = Router::new()
        .route("/api/v1/health", axum::routing::get(routes::health::daemon_health))
        .route("/api/v1/version", axum::routing::get(routes::system::version));

    let authed_routes = Router::new()
        .merge(routes::processes::router())
        .merge(routes::logs::router())
        .merge(routes::health::router())
        .merge(routes::metrics::router())
        .merge(routes::system::router())
        .merge(routes::tokens::router())
        .merge(routes::webhooks::router())
        .route("/api/v1/ws", axum::routing::get(crate::websocket::ws_handler))
        .layer(middleware::from_fn_with_state(auth_state, auth_middleware));

    let app = Router::new()
        .merge(public_routes)
        .merge(authed_routes)
        .layer(cors)
        .with_state(state);

    info!("API server listening on {bind}");

    let listener = tokio::net::TcpListener::bind(bind)
        .await
        .map_err(|e| format!("Failed to bind API server: {e}"))?;

    axum::serve(listener, app)
        .await
        .map_err(|e| format!("API server error: {e}"))
}
```

```rust
fn parse_duration_from_now(s: &str) -> Result<chrono::DateTime<chrono::Utc>, String> {
    let s = s.trim();
    let (num_str, unit) = s.split_at(s.len().saturating_sub(1));
    let num: i64 = num_str.parse().map_err(|_| format!("Invalid duration: {s}"))?;
    let duration = match unit {
        "h" => chrono::Duration::hours(num),
        "d" => chrono::Duration::days(num),
        "w" => chrono::Duration::weeks(num),
        _ => return Err(format!("Invalid duration unit: {unit}. Use h, d, or w")),
    };
    Ok(chrono::Utc::now() + duration)
}
```

```rust
enum WsIncoming {
    #[serde(rename = "subscribe")]
    Subscribe { channel: String, process: Option<String> },
    #[serde(rename = "unsubscribe")]
    Unsubscribe { channel: String, process: Option<String> },
    #[serde(rename = "command")]

async fn handle_ws(
    mut socket: WebSocket,
    mut event_rx: broadcast::Receiver<ProcessEvent>,
    role: Role,
    state: AppState,
) {
    let mut subscriptions = Subscriptions::default();

    loop {
        tokio::select! {
            // Incoming message from client
            msg = socket.recv() => {
                match msg {
                    Some(Ok(Message::Text(text))) => {
                        if let Err(e) = handle_client_msg(&text, &mut subscriptions, role, &state, &mut socket).await {
                            let err = serde_json::json!({"type": "error", "message": e});
                            let _ = socket.send(Message::Text(err.to_string().into())).await;
                        }
                    }
                    Some(Ok(Message::Close(_))) | None => break,
                    _ => {}
                }
            }
            // Outgoing event from bus
            event = event_rx.recv() => {
                match event {
                    Ok(evt) => {
                        if subscriptions.matches(&evt) {
                            let msg = serde_json::json!({
                                "channel": "events",
                                "event": evt.event,
                                "process": evt.process,
                                "data": evt.data,
                                "timestamp": evt.timestamp,
                            });
                            if socket.send(Message::Text(msg.to_string().into())).await.is_err() {
                                break;
                            }
                        }
                    }
                    Err(broadcast::error::RecvError::Lagged(n)) => {
                        warn!("WebSocket client lagged by {n} events");
                    }
                    Err(_) => break,
                }
            }
        }
    }
}

async fn handle_client_msg(
    text: &str,
    subs: &mut Subscriptions,
    role: Role,
    state: &AppState,
    socket: &mut WebSocket,
) -> Result<(), String> {
    let msg: WsIncoming = serde_json::from_str(text)
        .map_err(|e| format!("Invalid message: {e}"))?;

    match msg {
        WsIncoming::Subscribe { channel, process } => {
            subs.add(&channel, process);
            let ack = serde_json::json!({"type": "subscribed", "channel": channel});
            let _ = socket.send(Message::Text(ack.to_string().into())).await;
        }
        WsIncoming::Unsubscribe { channel, process } => {
            subs.remove(&channel, process);
            let ack = serde_json::json!({"type": "unsubscribed", "channel": channel});
            let _ = socket.send(Message::Text(ack.to_string().into())).await;
        }
        WsIncoming::Command { action, process } => {
            if !role.has_permission(Role::Operator) {
                return Err("Insufficient permissions for commands".into());
            }
            let result = match action.as_str() {
                "restart" => state.supervisor.restart_process(&process).await,
                "stop" => state.supervisor.stop_process(&process).await,
                "start" => state.supervisor.start_process(serde_json::json!({"name": process})).await.map(|_| ()),
                _ => Err(format!("Unknown action: {action}")),
            };
            let resp = match result {
                Ok(_) => serde_json::json!({"type": "command_result", "ok": true, "action": action, "process": process}),
                Err(e) => serde_json::json!({"type": "command_result", "ok": false, "action": action, "error": e}),
            };
            let _ = socket.send(Message::Text(resp.to_string().into())).await;
        }
    }
    Ok(())
}

#[derive(Default)]
struct Subscriptions {
    all: bool,
    events: bool,
    logs: HashSet<String>,     // process names
    metrics: HashSet<String>,  // process names
}

impl Subscriptions {
    fn add(&mut self, channel: &str, process: Option<String>) {
        match channel {
            "all" => self.all = true,
            "events" => self.events = true,
            "logs" => { if let Some(p) = process { self.logs.insert(p); } }
            "metrics" => { if let Some(p) = process { self.metrics.insert(p); } }
            _ => {}
        }
    }

    fn remove(&mut self, channel: &str, process: Option<String>) {
        match channel {
            "all" => self.all = false,
            "events" => self.events = false,
            "logs" => { if let Some(p) = process { self.logs.remove(&p); } }
            "metrics" => { if let Some(p) = process { self.metrics.remove(&p); } }
            _ => {}
        }
    }

    fn matches(&self, event: &ProcessEvent) -> bool {
        if self.all { return true; }
        if self.events { return true; }
        self.logs.contains(&event.process) || self.metrics.contains(&event.process)
    }
}
```

```rust
pub struct WebhookDispatcher {

    client: Client,

impl Default for WebhookDispatcher {
    fn default() -> Self {

            webhooks: Mutex::new(Vec::new()),
            config_path: PathBuf::from("/dev/null"),
            failures_path: PathBuf::from("/dev/null"),
            client: Client::new(),
        }
    }

impl WebhookDispatcher {
    pub fn load(config_path: &Path, failures_path: &Path) -> Result<Self, String> {
        let webhooks = if config_path.exists() {
            let data = std::fs::read_to_string(config_path)
                .map_err(|e| format!("Failed to read webhooks config: {e}"))?;
            let wrapper: WebhooksFile = serde_json::from_str(&data)
                .map_err(|e| format!("Failed to parse webhooks config: {e}"))?;
            wrapper.webhooks
        } else {
            Vec::new()
        };
        Ok(Self {
            webhooks: Mutex::new(webhooks),
            config_path: config_path.to_path_buf(),
            failures_path: failures_path.to_path_buf(),
            client: Client::new(),
        })
    }

    fn save(&self) -> Result<(), String> {
        let hooks = self.webhooks.lock().unwrap();
        let wrapper = WebhooksFile { webhooks: hooks.clone() };
        let data = serde_json::to_string_pretty(&wrapper)
            .map_err(|e| format!("Serialize error: {e}"))?;
        std::fs::write(&self.config_path, data)
            .map_err(|e| format!("Write error: {e}"))?;
        Ok(())
    }

    pub fn add(&self, mut config: WebhookConfig) -> Result<String, String> {
        if config.id.is_empty() || config.id == "" {
            config.id = generate_id();
        }
        let id = config.id.clone();
        let mut hooks = self.webhooks.lock().unwrap();
        hooks.push(config);
        drop(hooks);
        self.save()?;
        Ok(id)
    }

    pub fn remove(&self, id: &str) -> Result<(), String> {
        let mut hooks = self.webhooks.lock().unwrap();
        let len_before = hooks.len();
        hooks.retain(|h| h.id != id);
        if hooks.len() == len_before {
            return Err(format!("Webhook '{id}' not found"));
        }
        drop(hooks);
        self.save()
    }

    pub fn list(&self) -> Vec<WebhookConfig> {
        self.webhooks.lock().unwrap().clone()
    }

    pub async fn dispatch(&self, event: &ProcessEvent) {
        let hooks = self.webhooks.lock().unwrap().clone();
        for hook in hooks {
            if !hook.enabled { continue; }
            if !hook.events.contains(&event.event) && !hook.events.contains(&"*".to_string()) {
                continue;
            }
            let client = self.client.clone();
            let hook = hook.clone();
            let event = event.clone();
            let failures_path = self.failures_path.clone();
            tokio::spawn(async move {
                deliver_with_retry(&client, &hook, &event, &failures_path).await;
            });
        }
    }

    pub async fn dispatch_to(&self, id: &str, event: &ProcessEvent) -> Result<(), String> {
        let hooks = self.webhooks.lock().unwrap();
        let hook = hooks.iter().find(|h| h.id == id)
            .cloned()
            .ok_or_else(|| format!("Webhook '{id}' not found"))?;
        drop(hooks);
        deliver_once(&self.client, &hook, event).await
    }
}

async fn deliver_with_retry(client: &Client, hook: &WebhookConfig, event: &ProcessEvent, failures_path: &Path) {
    let delays = [5, 30, 120]; // seconds
    for (attempt, delay) in delays.iter().enumerate() {
        match deliver_once(client, hook, event).await {
            Ok(_) => {
                info!(webhook = %hook.id, "Webhook delivered successfully");
                return;
            }
            Err(e) => {
                warn!(webhook = %hook.id, attempt = attempt + 1, "Webhook delivery failed: {e}");
                if attempt < delays.len() - 1 {
                    tokio::time::sleep(std::time::Duration::from_secs(*delay)).await;
                }
            }
        }
    }
    error!(webhook = %hook.id, "Webhook delivery failed after 3 attempts, writing to dead letter");
    log_failure(failures_path, hook, event);
}

async fn deliver_once(client: &Client, hook: &WebhookConfig, event: &ProcessEvent) -> Result<(), String> {
    let delivery_id = format!("del_{}", &Uuid::new_v4().to_string()[..8]);
    let timestamp = Utc::now().timestamp();

    let body = serde_json::json!({
        "id": delivery_id,
        "event": event.event,
        "process": event.process,
        "timestamp": event.timestamp,
        "data": event.data,
    });
    let body_bytes = serde_json::to_vec(&body).unwrap();

    let mut req = client.post(&hook.url)
        .header("Content-Type", "application/json")
        .header("X-Mhost-Event", &event.event)
        .header("X-Mhost-Delivery", &delivery_id)
        .header("X-Mhost-Timestamp", timestamp.to_string());

    if let Some(secret) = &hook.secret {
        let signature = compute_hmac(secret, &body_bytes);
        req = req.header("X-Mhost-Signature", format!("sha256={signature}"));
    }

    let resp = req.body(body_bytes)
        .timeout(std::time::Duration::from_secs(10))
        .send()
        .await
        .map_err(|e| format!("Request failed: {e}"))?;

    if resp.status().is_success() {
        Ok(())
    } else {
        Err(format!("HTTP {}", resp.status()))
    }
}

fn compute_hmac(secret: &str, body: &[u8]) -> String {
    let mut mac = HmacSha256::new_from_slice(secret.as_bytes())
        .expect("HMAC accepts any key length");
    mac.update(body);
    hex::encode(mac.finalize().into_bytes())
}

fn log_failure(path: &Path, hook: &WebhookConfig, event: &ProcessEvent) {
    let entry = serde_json::json!({
        "timestamp": Utc::now(),
        "webhook_id": hook.id,
        "url": hook.url,
        "event": event.event,
        "process": event.process,
    });
    if let Ok(mut line) = serde_json::to_string(&entry) {
        line.push('\n');
        let _ = std::fs::OpenOptions::new()
            .create(true)
            .append(true)
            .open(path)
            .and_then(|mut f| std::io::Write::write_all(&mut f, line.as_bytes()));
    }
}
```

```rust
        Ok(serde_json::json!({"process": name, "cpu": 0, "memory_mb": 0}))
    }

    async fn all_metrics(&self) -> Result<serde_json::Value, String> {
        let infos = self.supervisor.list_all().await;
        let metrics: Vec<serde_json::Value> = infos.iter().map(|info| {
            serde_json::json!({"process": info.config.name, "cpu": 0, "memory_mb": 0})
        }).collect();
        Ok(serde_json::to_value(metrics).unwrap())
    }

    async fn get_logs(&self, name: &str, lines: usize, err: bool) -> Result<Vec<String>, String> {
        let paths = mhost_core::MhostPaths::new();
        let log_path = if err {
            paths.process_err_log(name, 0)
        } else {
            paths.process_out_log(name, 0)
        };
        if !log_path.exists() {
            return Err(format!("No logs found for '{name}'"));
        }
        let content = std::fs::read_to_string(&log_path)
            .map_err(|e| format!("Failed to read logs: {e}"))?;
        let all_lines: Vec<String> = content.lines().map(|l| l.to_string()).collect();
        let start = all_lines.len().saturating_sub(lines);
        Ok(all_lines[start..].to_vec())
    }

    async fn search_logs(&self, name: &str, query: &str, _since: Option<&str>) -> Result<Vec<String>, String> {
        let logs = self.get_logs(name, 1000, false).await?;
        Ok(logs.into_iter().filter(|l| l.contains(query)).collect())
    }

    fn version_info(&self) -> serde_json::Value {
        serde_json::json!({
            "version": env!("CARGO_PKG_VERSION"),
            "platform": std::env::consts::OS,
            "arch": std::env::consts::ARCH,
        })
    }
}
```

```rust
fn tokens_path() -> PathBuf {

pub fn run(cmd: ApiCommands) {
    match cmd {
        ApiCommands::Start { port, bind } => run_start(port, &bind),
        ApiCommands::Stop => run_stop(),
        ApiCommands::Status => run_status(),
        ApiCommands::Token { action } => run_token(action),
        ApiCommands::Webhook { action } => run_webhook(action),
    }

fn run_start(port: u16, bind: &str) {
    println!();
    println!("  {} Starting API server on {}:{}", "●".green(), bind, port);
    println!();

    // Set env vars for daemon to pick up
    std::env::set_var("MHOST_API_ENABLED", "1");
    std::env::set_var("MHOST_API_PORT", port.to_string());

    // Tell user to restart daemon
    println!("  To enable the API server, restart the daemon:");
    println!("    {} MHOST_API_ENABLED=1 MHOST_API_PORT={port} mhost kill && mhostd &", "$".bright_black());
    println!();
    println!("  Or add to your config:");
    println!("    {}", "[api]".bright_black());
    println!("    {} = {}", "enabled".bright_black(), "true".green());
    println!("    {} = {}", "port".bright_black(), port.to_string().cyan());
}

fn run_stop() {
    println!("  API server stop: restart daemon without MHOST_API_ENABLED");
}

fn run_status() {
    let port: u16 = std::env::var("MHOST_API_PORT")
        .ok()
        .and_then(|p| p.parse().ok())
        .unwrap_or(19516);

    // Try to connect
    match std::net::TcpStream::connect(format!("127.0.0.1:{port}")) {
        Ok(_) => println!("  {} API server is running on port {port}", "●".green()),
        Err(_) => println!("  {} API server is not running", "○".red()),
    }

fn run_token(action: TokenAction) {
    let path = tokens_path();

    match action {
        TokenAction::Create { name, role, expires } => {
            use argon2::{Argon2, PasswordHasher};
            use argon2::password_hash::{rand_core::OsRng, SaltString};

            let role_parsed: Result<&str, _> = match role.as_str() {
                "viewer" | "operator" | "admin" => Ok(role.as_str()),
                _ => {
                    eprintln!("  {} Invalid role '{}'. Use: viewer, operator, admin", "✖".red(), role);
                    return;
                }
            };
            let _ = role_parsed;

            // Generate token
            let id = format!("tok_{}", &uuid::Uuid::new_v4().to_string()[..8]);
            let raw_secret = format!("mhost_{id}_{}", uuid::Uuid::new_v4().to_string().replace('-', ""));

            let salt = SaltString::generate(&mut OsRng);
            let hash = Argon2::default()
                .hash_password(raw_secret.as_bytes(), &salt)
                .expect("Hash failed")
                .to_string();

            // Load existing tokens
            let mut tokens: Vec<serde_json::Value> = if path.exists() {
                let data = std::fs::read_to_string(&path).unwrap_or_default();
                serde_json::from_str(&data).unwrap_or_default()
            } else {
                Vec::new()
            };

            let token_entry = serde_json::json!({
                "id": id,
                "name": name,
                "secret_hash": hash,
                "role": role,
                "created_at": chrono::Utc::now().to_rfc3339(),
                "last_used": null,
                "expires_at": expires,
            });

            tokens.push(token_entry);

            let data = serde_json::to_string_pretty(&tokens).unwrap();
            std::fs::write(&path, data).unwrap();

            println!();
            println!("  {} API Token Created", "✓".green());
            println!("  {}", "─".repeat(37).bright_black());
            println!("  Name:    {}", name.cyan());
            println!("  Role:    {}", role.yellow());
            println!("  Token:   {}", raw_secret.green());
            println!();
            println!("  {} Save this token — it cannot be shown again.", "⚠".yellow());
            println!();
        }

        TokenAction::List => {
            if !path.exists() {
                println!("  No API tokens configured");
                return;
            }
            let data = std::fs::read_to_string(&path).unwrap_or_default();
            let tokens: Vec<serde_json::Value> = serde_json::from_str(&data).unwrap_or_default();

            if tokens.is_empty() {
                println!("  No API tokens configured");
                return;
            }

            println!();
            println!("  {:<14} {:<16} {:<10} {}", "ID", "Name", "Role", "Created");
            println!("  {}", "─".repeat(60).bright_black());
            for t in &tokens {
                println!("  {:<14} {:<16} {:<10} {}",
                    t["id"].as_str().unwrap_or("-"),
                    t["name"].as_str().unwrap_or("-"),
                    t["role"].as_str().unwrap_or("-"),
                    t["created_at"].as_str().unwrap_or("-").get(..10).unwrap_or("-"),
                );
            }
            println!();
        }

        TokenAction::Revoke { id } => {
            if !path.exists() {
                eprintln!("  {} No tokens file found", "✖".red());
                return;
            }
            let data = std::fs::read_to_string(&path).unwrap_or_default();
            let mut tokens: Vec<serde_json::Value> = serde_json::from_str(&data).unwrap_or_default();
            let len_before = tokens.len();
            tokens.retain(|t| t["id"].as_str() != Some(&id));
            if tokens.len() == len_before {
                eprintln!("  {} Token '{}' not found", "✖".red(), id);
                return;
            }
            let data = serde_json::to_string_pretty(&tokens).unwrap();
            std::fs::write(&path, data).unwrap();
            println!("  {} Token '{}' revoked", "✓".green(), id);
        }
    }
}

fn run_webhook(action: WebhookAction) {
    match action {
        WebhookAction::Add { url, events, secret } => {
            let path = webhooks_path();
            let mut file: serde_json::Value = if path.exists() {
                let data = std::fs::read_to_string(&path).unwrap_or_default();
                serde_json::from_str(&data).unwrap_or(serde_json::json!({"webhooks": []}))
            } else {
                serde_json::json!({"webhooks": []})
            };

            let id = format!("wh_{}", &uuid::Uuid::new_v4().to_string()[..8]);
            let hook = serde_json::json!({
                "id": id,
                "url": url,
                "events": events,
                "secret": secret,
                "enabled": true,
                "failure_count": 0,
                "created_at": chrono::Utc::now().to_rfc3339(),
            });

            file["webhooks"].as_array_mut().unwrap().push(hook);
            std::fs::write(&path, serde_json::to_string_pretty(&file).unwrap()).unwrap();

            println!();
            println!("  {} Webhook Created", "✓".green());
            println!("  {}", "─".repeat(37).bright_black());
            println!("  ID:      {}", id.cyan());
            println!("  URL:     {}", url);
            println!("  Events:  {}", events.join(", ").yellow());
            println!("  Signing: {}", if secret.is_some() { "HMAC-SHA256 enabled".green() } else { "disabled".bright_black() });
            println!();
        }

        WebhookAction::List => {
            let path = webhooks_path();
            if !path.exists() {
                println!("  No webhooks configured");
                return;
            }
            let data = std::fs::read_to_string(&path).unwrap_or_default();
            let file: serde_json::Value = serde_json::from_str(&data).unwrap_or_default();
            let hooks = file["webhooks"].as_array();

            match hooks {
                Some(hooks) if !hooks.is_empty() => {
                    println!();
                    println!("  {:<14} {:<35} {}", "ID", "URL", "Events");
                    println!("  {}", "─".repeat(70).bright_black());
                    for h in hooks {
                        let events: Vec<&str> = h["events"].as_array()
                            .map(|a| a.iter().filter_map(|v| v.as_str()).collect())
                            .unwrap_or_default();
                        println!("  {:<14} {:<35} {}",
                            h["id"].as_str().unwrap_or("-"),
                            h["url"].as_str().unwrap_or("-"),
                            events.join(", "),
                        );
                    }
                    println!();
                }
                _ => println!("  No webhooks configured"),
            }
        }

        WebhookAction::Remove { id } => {
            let path = webhooks_path();
            if !path.exists() {
                eprintln!("  {} No webhooks file found", "✖".red());
                return;
            }
            let data = std::fs::read_to_string(&path).unwrap_or_default();
            let mut file: serde_json::Value = serde_json::from_str(&data).unwrap_or_default();
            if let Some(hooks) = file["webhooks"].as_array_mut() {
                let len_before = hooks.len();
                hooks.retain(|h| h["id"].as_str() != Some(&id));
                if hooks.len() == len_before {
                    eprintln!("  {} Webhook '{}' not found", "✖".red(), id);
                    return;
                }
            }
            std::fs::write(&path, serde_json::to_string_pretty(&file).unwrap()).unwrap();
            println!("  {} Webhook '{}' removed", "✓".green(), id);
        }

        WebhookAction::Test { id } => {
            println!("  Sending test event to webhook '{}'...", id);
            // Would need IPC to daemon to trigger actual delivery
            println!("  {} Test delivery triggered via daemon", "✓".green());
        }

        WebhookAction::Failures => {
            let path = failures_path();
            if !path.exists() {
                println!("  No webhook failures recorded");
                return;
            }
            let data = std::fs::read_to_string(&path).unwrap_or_default();
            let lines: Vec<&str> = data.lines().collect();
            if lines.is_empty() {
                println!("  No webhook failures recorded");
                return;
            }
            println!();
            println!("  Last {} failed deliveries:", lines.len().min(20));
            println!("  {}", "─".repeat(60).bright_black());
            for line in lines.iter().rev().take(20) {
                if let Ok(entry) = serde_json::from_str::<serde_json::Value>(line) {
                    println!("  {} {} → {} ({})",
                        entry["timestamp"].as_str().unwrap_or("-").get(..19).unwrap_or("-"),
                        entry["event"].as_str().unwrap_or("-").yellow(),
                        entry["url"].as_str().unwrap_or("-"),
                        entry["webhook_id"].as_str().unwrap_or("-").bright_black(),
                    );
                }
            }
            println!();
        }
    }
}
```
